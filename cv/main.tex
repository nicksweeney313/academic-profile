% --- Sort newest first by full date (YYYY-MM-DD), then author, then title
\DeclareSortingScheme{dateydnt}{%
  \sort[direction=descending]{%
    \field{sortdate}%
    \field{date}%
    \field{year}%
  }%
  \sort{\name{author}}%
  \sort{\field{title}}%
}

  \field{date}
  \field{year}
}

}

% Strip "https://doi.org/" if you accidentally include it in the .bib
\DeclareSourcemap{
  \maps[datatype=bibtex]{
    \map{
      \step[fieldsource=doi,
            match=\regexp{^https?://(dx\.)?doi\.org/},
            replace={}]
    }
  }
}
